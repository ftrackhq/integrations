13def track_framework_usage(event_name, metadata, tracked_args=[]): 

14 """ 

15 Decorator to track usage of framework functions. 

16 

17 This decorator wraps a function to track its usage along with specified metadata and arguments. It ensures that a 

18 UsageTracker instance is set and available. If not, it attempts to initialize a default one using the session 

19 attribute of the first argument of the wrapped function, which is expected to be an instance of a class with a 

20 session attribute. 

21 

22 Parameters: 

23 *event_name* (str): The name of the event to track. 

24 *metadata* (dict): A dictionary of metadata to track along with the event. This can include any relevant 

25 information about the event. 

26 *tracked_args* (list): A list of argument names (as strings) of the wrapped function whose values should be 

27 included in the metadata. Defaults to an empty list. 

28 

29 The decorator updates the metadata dictionary with the values of the specified arguments (if present) before 

30 tracking the event. It logs a warning if a specified argument name does not exist in the function's arguments. 

31 

32 If the UsageTracker cannot be initialized (e.g., due to the absence of a session attribute), it logs an error 

33 and skips tracking. 

34 

35 Usage: 

36 @track_framework_usage('event_name', {'key': 'value'}, ['arg1', 'arg2']) 

37 def some_function(arg1, arg2): 

38 pass 

39 

40 This will track the usage of `some_function`, including the values of `arg1` and `arg2` in the metadata. 

41 

42 Returns: 

43 The original function wrapped with usage tracking functionality. 

44 """ 

45 

46 def decorator(func): 

47 @functools.wraps(func) 

48 def wrapper(*args, **kwargs): 

49 from ftrack_utils.usage import ( 

50 get_usage_tracker, 

51 set_usage_tracker, 

52 UsageTracker, 

53 ) 

54 

55 # Ensure UsageTracker is set and available 

56 usage_tracker = get_usage_tracker() 

57 if not usage_tracker: 

58 logger.warning( 

59 "UsageTracker instance is not set. Try to initialize a default one." 

60 ) 

61 # Try to instantiate UsageTracker if class has session 

62 try: 

63 if args and hasattr(args[0], '__class__'): 

64 instance = args[0] 

65 if hasattr(instance, 'session'): 

66 session = getattr(instance, 'session') 

67 else: 

68 return AttributeError( 

69 "The 'session' property is required to initialize UsageTracker but " 

70 "not found in the class. Skipping initialization as can't track usage." 

71 ) 

72 

73 # Determine the module and method name 

74 full_module_name = func.__module__ 

75 # Extract the root module name 

76 root_module_name = full_module_name.split('.')[0] 

77 method_name = func.__name__ 

78 

79 # Retrieve the root module and its version 

80 module = sys.modules[root_module_name] 

81 

82 library_version = '0.0.0' 

83 if hasattr(module, '__version__'): 

84 library_version = module.__version__ 

85 else: 

86 logger.warning( 

87 f"The root module {root_module_name} does not have a " 

88 f"'__version__' attribute. Setting 0.0.0 as default version." 

89 ) 

90 

91 # Get the current OS platform 

92 current_os = platform.system() 

93 

94 # Setup metadata 

95 default_metadata = dict( 

96 module_name=root_module_name, 

97 version=library_version, 

98 os=current_os, 

99 ) 

100 # Set the Usage tracker 

101 set_usage_tracker( 

102 UsageTracker( 

103 session=session, default_data=default_metadata 

104 ) 

105 ) 

106 

107 except Exception as e: 

108 logger.error(e) 

109 

110 finally: 

111 usage_tracker = get_usage_tracker() 

112 

113 # Get the function's argument names and values 

114 func_args = inspect.signature(func).parameters 

115 arg_names = list(func_args.keys()) 

116 

117 # Build a dictionary of argument names and values 

118 arg_values = args + tuple(kwargs.values()) 

119 args_metadata = dict(zip(arg_names, arg_values)) 

120 

121 # Update metadata with function arguments if specified in kwarg_mapping_list 

122 for key in tracked_args: 

123 if key in args_metadata: 

124 metadata[key] = args_metadata[key] 

125 else: 

126 logger.warning( 

127 f'Provided kwarg mapping {key} seems to not ' 

128 f'exists as argument of this function' 

129 ) 

130 

131 if usage_tracker: 

132 usage_tracker.track(event_name, metadata) 

133 

134 # Call the original function 

135 return func(*args, **kwargs) 

136 

137 return wrapper 

138 

139 return decorator
